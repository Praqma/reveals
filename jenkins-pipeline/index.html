Jenkinsfile (declarative) example 


						pipeline {
	agent any

	stages {
		stage('build'){
			steps {
				echo 'this is the build stage'
				script {
					make
				}
			}
		}
		stage('test'){
			echo 'this is the test stage'
		}
		stage('deploy'){
			when ( branch 'production')
			echo 'this is the deploy stage'
		}
	}
}
